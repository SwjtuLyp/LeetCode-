#include<iostream>
#include<cstdio>
#include<string.h>
#include<algorithm>
#include<stdlib.h>
#include<queue>
#include<vector>
using namespace std;

int main(){
    vector< vector<int> > g;
    int n,m;
    scanf("%d %d",&n,&m);
    vector<int >color(n+1,0);
    //0ÎÞÉ«£¬1ºÚ-1°×
    for(int i=0;i<n+1;i++){
        vector<int>x;
        g.push_back(x);
    }
    for(int i=0;i<m;i++){
        int u,v;
        scanf("%d %d",&u,&v);
        g[u].push_back(v);
        g[v].push_back(u);
    }
    int i=1;
    while(i<n+1){
        if(color[i]==0){
            color[i]=1;
            queue<int >q;
            q.push(i);
            while(!q.empty()){
                int t=q.front();
                q.pop();
                int k=0;
                while(k<g[t].size()){
                    if(color[t]==color[g[t][k]]){
                        cout<<"No";
                        return 0;
                    }else{
                        if(color[g[t][k]]==0)
                            q.push(g[t][k]);
                        color[g[t][k]]=-color[t];
                    }
                   k++;
                }
            }
        }

        i++;
    }
    cout<<"Yes";
    return 0;
}
